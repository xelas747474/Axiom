const rsi=(c,p=14)=>{if(c.length<p+1)return 50;let g=0,l=0;for(let i=c.length-p;i<c.length;i++){const d=c[i]-c[i-1];d>0?g+=d:l-=d;}return 100-(100/(1+(g/p)/((l/p)||.001)));};

const macd=c=>{const m=ema(c,12).map((v,i)=>v-ema(c,26)[i]);const s=ema(m,9);return{m,s,h:m.map((v,i)=>v-s[i])};};
const bb=(c,p=20,n=2)=>c.map((_,i)=>{const sl=c.slice(Math.max(0,i-p+1),i+1),mn=sl.reduce((a,b)=>a+b,0)/sl.length,sd=Math.sqrt(sl.reduce((a,b)=>a+(b-mn)**2,0)/sl.length);return{u:mn+n*sd,m:mn,l:mn-n*sd};});
const sr=(c,r=20)=>{const l=c.slice(-r),hi=Math.max(...l),lo=Math.min(...l),vw=l.reduce((a,b)=>a+b,0)/l.length;return{r2:hi*1.025,r1:hi*1.008,s1:lo*.992,s2:lo*.975,vw};};
const fib=c=>{const s=c.slice(-30),hi=Math.max(...s),lo=Math.min(...s),d=hi-lo;return[{l:'0.000',p:hi},{l:'0.236',p:hi-d*.236},{l:'0.382',p:hi-d*.382},{l:'0.500',p:hi-d*.5},{l:'0.618',p:hi-d*.618},{l:'0.786',p:hi-d*.786},{l:'1.000',p:lo}];};
const struct=c=>{const ev=[];if(c.length<8)return ev;const pv=c.slice(-8,-4),la=c.slice(-4),pH=Math.max(...pv),pL=Math.min(...pv),lH=Math.max(...la),lL=Math.min(...la),pr=c[c.length-1];if(lH>pH&&lL>pL)ev.push({t:'HH',l:'Higher High',p:lH});if(lL>pL)ev.push({t:'HL',l:'Higher Low',p:lL});if(lH<pH&&lL<pL)ev.push({t:'LL',l:'Lower Low',p:lL});if(lH<pH)ev.push({t:'LH',l:'Lower High',p:lH});if(pr>pH*1.002)ev.push({t:'BOS',l:'Break of Structure',p:pr});if(pH>lH&&pr<pH*.998&&lL<pL)ev.push({t:'CHOCH',l:'CHoCH Detected',p:pr});return ev.slice(0,5);};

}

function drawMain(d,id){
  const canvas=document.getElementById('mainChart');
  const W=canvas.parentElement.clientWidth-24,H=200;
  canvas.width=W;canvas.height=H;
  const ctx=canvas.getContext('2d');ctx.clearRect(0,0,W,H);
  const c=d.closes,n=c.length,bv=bb(c);
  const mn=Math.min(...bv.map(b=>b.l),d.sr.s2)*.997,mx=Math.max(...bv.map(b=>b.u),d.sr.r2)*1.003;
  const py=v=>((mx-v)/(mx-mn))*H,px=i=>(i/(n-1))*W;

  ctx.strokeStyle='rgba(255,255,255,.025)';ctx.lineWidth=1;
  for(let i=1;i<4;i++){ctx.beginPath();ctx.moveTo(0,H/4*i);ctx.lineTo(W,H/4*i);ctx.stroke();}

  ctx.beginPath();bv.forEach((b,i)=>i===0?ctx.moveTo(px(i),py(b.u)):ctx.lineTo(px(i),py(b.u)));
  bv.slice().reverse().forEach((b,i)=>ctx.lineTo(px(n-1-i),py(b.l)));
  ctx.closePath();ctx.fillStyle='rgba(41,128,185,0.05)';ctx.fill();

  const dl=(pts,col,dash=[],w=1)=>{ctx.beginPath();ctx.strokeStyle=col;ctx.lineWidth=w;ctx.setLineDash(dash);pts.forEach((p,i)=>i===0?ctx.moveTo(p[0],p[1]):ctx.lineTo(p[0],p[1]));ctx.stroke();ctx.setLineDash([]);};
  dl(bv.map((b,i)=>[px(i),py(b.u)]),'rgba(41,128,185,.35)',[3,3]);

function drawMacd(d){
  const canvas=document.getElementById('macdChart');

  canvas.width=W;canvas.height=H;
  const ctx=canvas.getContext('2d');ctx.clearRect(0,0,W,H);
  const h=d.macdH,n=h.length,mx=Math.max(...h.map(Math.abs))||1,zero=H/2,bw=W/n*.75;
  h.forEach((v,i)=>{const x=(i/n)*W,ht=Math.abs(v)/mx*(H/2-2),bull=v>=0;ctx.fillStyle=bull?'rgba(45,158,95,.6)':'rgba(192,57,43,.6)';ctx.fillRect(x,bull?zero-ht:zero,bw,ht);});

  const dl=(arr,col)=>{ctx.beginPath();ctx.strokeStyle=col;ctx.lineWidth=1.5;arr.forEach((v,i)=>{const x=(i/n)*W,y=zero-v/mx*(H/2-2);i===0?ctx.moveTo(x,y):ctx.lineTo(x,y);});ctx.stroke();};
  dl(d.macdArr,'rgba(41,128,185,.75)');dl(d.sigArr,'rgba(201,168,76,.85)');

// ── SIGNAL ENGINE ──────────────────────────────────────
function computeSignals(id){
  const d=mkt[id];if(!d||d.closes.length<30)return[];
  const closes=d.closes,n=closes.length,bv=bb(closes),fv=fib(closes),stv=struct(closes);
  const mc=macd(closes);
  const signals=[];
  const scanRange=Math.min(20,n-2);

  for(let i=n-scanRange;i<n;i++){
    const pi=closes[i],bi=bv[i];
    const ri=rsi(closes.slice(0,i+1));
    const mi=mc.m[i],si=mc.s[i],mp=mc.m[i-1],sp=mc.s[i-1];
    const isCur=(i===n-1);
    const candleOffset=n-1-i;
    const tLabel=candleOffset===0?'Maintenant':'Il y a '+candleOffset+' bougies';

    // BUY confluences
    const bc=[];
    if(ri<38)bc.push({tag:'RSI Survente '+ri.toFixed(0),str:ri<30?35:20});
    if(pi<=bi.l*1.006)bc.push({tag:'BB Lower Touch',str:25});
    if(mp<sp&&mi>=si)bc.push({tag:'MACD Cross ▲',str:30});
    const nfib=fv.slice(2,6).find(f=>Math.abs(pi-f.p)/pi<0.013);
    if(nfib)bc.push({tag:'Fib '+nfib.l,str:20});
    if(stv.some(s=>s.t==='HL'||s.t==='HH')&&isCur)bc.push({tag:'Structure HL',str:25});
    if(pi<d.vwap*.994)bc.push({tag:'Sous VWAP',str:15});

    // SELL confluences
    const sc=[];
    if(ri>62)sc.push({tag:'RSI Surachat '+ri.toFixed(0),str:ri>70?35:20});
    if(pi>=bi.u*.994)sc.push({tag:'BB Upper Touch',str:25});
    if(mp>sp&&mi<=si)sc.push({tag:'MACD Cross ▼',str:30});
    if(Math.abs(pi-d.sr.r1)/pi<0.011)sc.push({tag:'Résistance R1',str:25});
    if(Math.abs(pi-d.sr.r2)/pi<0.011)sc.push({tag:'Résistance R2',str:30});
    if(stv.some(s=>s.t==='LH'||s.t==='CHOCH')&&isCur)sc.push({tag:'LH/CHoCH',str:30});
    if(pi>d.vwap*1.016)sc.push({tag:'Au-dessus VWAP',str:15});

    const bScore=bc.reduce((a,c)=>a+c.str,0);
    const sScore=sc.reduce((a,c)=>a+c.str,0);
    const spread=bi.u-bi.l;

    if(bc.length>=2&&bScore>=40&&bScore>sScore){
      const tp1=pi+spread*.5,tp2=pi+spread*1.1,sl=Math.max(pi-spread*.4,d.sr.s2);
      signals.push({type:'BUY',asset:id.toUpperCase(),sigPrice:pi,score:Math.min(100,bScore),confs:bc,tp1,tp2,sl,rr:((tp1-pi)/(pi-sl)||1).toFixed(1),time:tLabel,isCur,chartIdx:i,reason:buildReason('BUY',bc,ri,pi,d)});
    } else if(sc.length>=2&&sScore>=40&&sScore>bScore){
      const tp1=pi-spread*.5,tp2=pi-spread*1.0,sl=Math.min(pi+spread*.35,d.sr.r2);
      signals.push({type:'SELL',asset:id.toUpperCase(),sigPrice:pi,score:Math.min(100,sScore),confs:sc,tp1,tp2,sl,rr:((pi-tp1)/(sl-pi)||1).toFixed(1),time:tLabel,isCur,chartIdx:i,reason:buildReason('SELL',sc,ri,pi,d)});
    }
  }

  // Deduplicate: max 1 per type per 3% price range
  const dedup=[];
  for(const s of signals.slice().reverse()){
    const dup=dedup.some(x=>x.type===s.type&&Math.abs(x.sigPrice-s.sigPrice)/s.sigPrice<0.03);
    if(!dup)dedup.unshift(s);
  }
  return dedup.slice(0,5);
}

function buildReason(type,confs,ri,price,d){
  const tags=confs.map(c=>c.tag);
  const parts=[];
  if(type==='BUY'){
    if(tags.some(t=>t.includes('RSI')))parts.push('RSI en survente ('+ri.toFixed(0)+')');
    if(tags.some(t=>t.includes('BB Lower')))parts.push('contact Bollinger Lower Band');
    if(tags.some(t=>t.includes('MACD')))parts.push('croisement MACD haussier');
    if(tags.some(t=>t.includes('Fib')))parts.push('rebond sur niveau Fibonacci');
    if(tags.some(t=>t.includes('Structure')))parts.push('structure Higher Low confirmée');
  }else{
    if(tags.some(t=>t.includes('RSI')))parts.push('RSI en surachat ('+ri.toFixed(0)+')');
    if(tags.some(t=>t.includes('BB Upper')))parts.push('contact Bollinger Upper Band');
    if(tags.some(t=>t.includes('MACD')))parts.push('croisement MACD baissier');
    if(tags.some(t=>t.includes('Résistance')))parts.push('rejet sur résistance majeure');
    if(tags.some(t=>t.includes('LH')||t.includes('CHoCH')))parts.push('structure Lower High / CHoCH détecté');
  }
  return parts.length?parts.join(', ')+'.' :'Confluence multi-indicateurs.';
}

  });

}
